### 🔍 重複判定アルゴリズム
```python
def check_article_duplication(new_request):
    """
    新しい記事作成要求の重複チェック
    """
    
    # 1. メインキーワード完全一致チェック
    main_keyword_check = check_main_keyword_exact_match(new_request)
    
    # 2. キーワード組み合わせ類似度チェック  
    keyword_similarity = calculate_keyword_similarity(new_request)
    
    # 3. ターゲット読者層重複チェック
    target_overlap = check_target_audience_overlap(new_request)
    
    # 4. 記事角度・切り口重複チェック
    angle_similarity = check_article_angle_similarity(new_request)
    
    # 5. プロジェクト横断重複チェック
    cross_project_check = check_cross_project_duplication(new_request)
    
    # 総合判定
    if main_keyword_check["exact_match"]:
        return {"status": "DUPLICATE", "reason": "メインキーワード完全重複"}
    elif keyword_similarity > 0.8:
        return {"status": "HIGH_RISK", "reason": "キーワード高類似", "suggestions": generate_differentiation_suggestions(new_request)}
    elif target_overlap and angle_similarity > 0.7:
        return {"status": "MODERATE_RISK", "reason": "読者層・角度類似", "suggestions": generate_angle_variations(new_request)}
    else:
        return {"status": "SAFE", "unique_points": identify_unique_elements(new_request)}
```

### 品質スコア判定基準
```python
def calculate_template_quality_score(article_data: Dict[str, Any]) -> Dict[str, Any]:
    """テンプレート準拠度とSEO品質のスコア算出"""
    
    score_breakdown = {
        "content_structure": 0,    # 構成要素の完備度
        "readability": 0,          # 読みやすさ
        "engagement": 0,           # エンゲージメント要素
        "seo_optimization": 0,     # SEO最適化
        "conversion_focus": 0      # コンバージョン導線
    }
    
    # 1. 構成要素チェック（30点満点）
    required_sections = [
        "この記事で分かること",
        "FAQ",
        "まとめ"
    ]
    
    for section in required_sections:
        if section in article_data["content"]:
            score_breakdown["content_structure"] += 10
    
    # 2. 読みやすさチェック（20点満点）
    readability_factors = {
        "conversational_tone": check_conversational_tone(article_data["content"]),
        "visual_elements": count_visual_elements(article_data["content"]),
        "paragraph_length": check_paragraph_length(article_data["content"])
    }
    
    score_breakdown["readability"] = sum(readability_factors.values()) * 20 / 3
    
    # 3. エンゲージメント要素（20点満点）
    engagement_elements = {
        "questions_to_reader": count_questions(article_data["content"]),
        "emotional_connection": check_emotional_words(article_data["content"]),
        "actionable_advice": count_actionable_items(article_data["content"])
    }
    
    score_breakdown["engagement"] = sum(engagement_elements.values()) * 20 / 3
    
    # 4. SEO最適化（15点満点）
    seo_factors = {
        "keyword_density": check_keyword_density(article_data),
        "meta_description": check_meta_description(article_data),
        "heading_structure": check_heading_hierarchy(article_data["content"])
    }
    
    score_breakdown["seo_optimization"] = sum(seo_factors.values()) * 15 / 3
    
    # 5. コンバージョン導線（15点満点）
    conversion_elements = {
        "clear_cta": check_call_to_action(article_data["content"]),
        "benefit_emphasis": check_benefit_statements(article_data["content"]),
        "urgency_creation": check_urgency_elements(article_data["content"])
    }
    
    score_breakdown["conversion_focus"] = sum(conversion_elements.values()) * 15 / 3
    
    total_score = sum(score_breakdown.values())
    
    return {
        "total_score": total_score,
        "breakdown": score_breakdown,
        "grade": get_quality_grade(total_score),
        "improvement_suggestions": generate_improvement_suggestions(score_breakdown)
    }
```
